  exista, afisati cate dintre cuvintele  din el sunt nume de utilizator existent.

N=0
for F in $*;do
	if [ -f $F ];then
		for W in `cat  $F`;do
			if grep -q '^$W:' /etc/passwd;then
				N=`expr $N + 1`
			fi 
		done

	fi
