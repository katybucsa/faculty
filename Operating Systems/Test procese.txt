Sa se scrie doua procese A si B care comunica prin pipe sau FIFO.
A ii trimite lui B cativa intregi si B ii trimite inapoi lui A suma lor.

#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>


int main(){
        int a2b[2],b2a[2],a[10],suma,i;
        pipe(a2b);pipe(b2a);
        int n=9;
        for(i=1;i<=9;i++)
                a[i]=i;
        if(fork()==0){//proces A
                close(a2b[0]);
                close(b2a[1]);
                write(a2b[1],&n,sizeof(int));
                for(i=1;i<=9;i++){
                        write(a2b[1],&a[i],sizeof(int));
                }
                read(b2a[0],&suma,sizeof(int));
                printf("Suma numerelor este:%d\n ",suma);
                close(a2b[1]);close(b2a[0]);
                exit(0);
        }
        if(fork()==0){//proces B
                int suma1=0,nr,m,j;
                close(a2b[1]);
                close(b2a[0]);
                read(a2b[0],&m,sizeof(int));
                for( j=1;j<=m;j++){
                        read(a2b[0],&nr,sizeof(int));
                        suma1+=nr;
                }
                close(a2b[0]);
                write(b2a[1],&suma1,sizeof(int));
                close(b2a[1]);
                exit(0);
        }
        close(a2b[0]);close(a2b[1]);close(b2a[0]);close(b2a[1]);
        wait(0);
        wait(0);
        return 0;
}
